## Overview

This guide shows a complete Next.js integration using the App Router with:
- A client component using the `usePhonelink` hook
- A server-side route handler for token verification

## Install

```bash
npm install phonelink
```

## Project structure

```
app/
├── verify/
│   └── page.tsx          # Start page with verify button
├── auth/
│   └── callback/
│       └── page.tsx      # Callback page that processes the result
└── api/
    └── verify-phone/
        └── route.ts      # Server-side token verification
```

## 1. Verify button page

Create a client component that renders the verify button. When clicked, it redirects to Phonelink.

```tsx title="app/verify/page.tsx"
"use client";

import { usePhonelink } from "phonelink/web";

export default function VerifyPage() {
  const { startAuth } = usePhonelink({
    clientId: process.env.NEXT_PUBLIC_PHONELINK_CLIENT_ID!,
    callbackUrl: `${process.env.NEXT_PUBLIC_APP_URL}/auth/callback`,
  });

  return (
    <div>
      <h1>Verify your phone number</h1>
      <button onClick={startAuth}>
        Verify Phone Number
      </button>
    </div>
  );
}
```

## 2. Callback page

After verification, Phonelink redirects back to this page with a `token` query parameter. The `usePhonelink` hook detects this automatically and sends it to the server.

```tsx title="app/auth/callback/page.tsx"
"use client";

import { usePhonelink } from "phonelink/web";
import { useEffect, useState } from "react";
import { useRouter } from "next/navigation";

export default function CallbackPage() {
  const { result, isCallback } = usePhonelink({
    clientId: process.env.NEXT_PUBLIC_PHONELINK_CLIENT_ID!,
    callbackUrl: `${process.env.NEXT_PUBLIC_APP_URL}/auth/callback`,
  });
  const [status, setStatus] = useState<"verifying" | "success" | "error">("verifying");
  const router = useRouter();

  useEffect(() => {
    if (!result) return;

    fetch("/api/verify-phone", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ token: result.token, nonce: result.nonce }),
    })
      .then((res) => {
        if (!res.ok) throw new Error("Verification failed");
        return res.json();
      })
      .then(() => {
        setStatus("success");
        router.push("/dashboard");
      })
      .catch(() => setStatus("error"));
  }, [result, router]);

  if (status === "error") {
    return <p>Verification failed. Please try again.</p>;
  }

  return <p>Verifying your phone number...</p>;
}
```

## 3. Server-side verification route

Create an API route that verifies the JWT token using the Phonelink JWKS endpoint.

```ts title="app/api/verify-phone/route.ts"
import { verifyPhonelinkToken } from "phonelink/server";
import { NextRequest, NextResponse } from "next/server";

export async function POST(request: NextRequest) {
  const { token, nonce } = await request.json();

  if (!token || !nonce) {
    return NextResponse.json(
      { error: "Missing token or nonce" },
      { status: 400 },
    );
  }

  try {
    const payload = await verifyPhonelinkToken(
      token,
      nonce,
      process.env.PHONELINK_CLIENT_ID!,
    );

    // payload.phone_e164 contains the verified phone number
    // Save it to your database, associate with user session, etc.

    return NextResponse.json({
      phone: payload.phone_e164,
      verified: payload.verified,
    });
  } catch (error) {
    console.error("Phone verification failed:", error);
    return NextResponse.json(
      { error: "Invalid or expired token" },
      { status: 401 },
    );
  }
}
```

## 4. Environment variables

Add the following to your `.env.local`:

```bash title=".env.local"
NEXT_PUBLIC_PHONELINK_CLIENT_ID=your-client-id
NEXT_PUBLIC_APP_URL=http://localhost:3000
PHONELINK_CLIENT_ID=your-client-id
```

Note that `PHONELINK_CLIENT_ID` (without `NEXT_PUBLIC_`) is used on the server side for audience verification. This keeps your client ID out of client bundles for the server check, though it's fine for it to be public in the `startAuth` call.

## Complete flow

1. User visits `/verify` and clicks "Verify Phone Number"
2. `usePhonelink` generates a nonce, stores it in `sessionStorage`, and redirects to `https://phone.link/auth`
3. User completes phone verification on Phonelink
4. Phonelink redirects to `/auth/callback?token=eyJ...`
5. `usePhonelink` detects the callback, extracts the token and nonce
6. The callback page sends both to `POST /api/verify-phone`
7. The server verifies the JWT signature, nonce, audience, and verification status
8. On success, the server returns the verified phone number
